**madplay音乐播放器**

<img src="14-项目整合.assets/image-20260316104728395.png" alt="image-20260316104728395" style="zoom:50%;" />

```C
void madplay_ctl(void)
{
    char *mp3_path[3]={"1.mp3", "2.mp3", "3.mp3"};

    int i =0;
    while(1)
    {
        int x, y;
        char madplay_cmd[56] = {0};
        enum DIR dir = get_touch_dir(&x, &y); //获取点击屏幕的坐标

        if(dir == CLICK)
        {
            if() //上一首
            {
                system("killall -9 madplay");//关闭原来播放的音乐
                i--;
                if(i < 0)
                    i = 2;
                sprintf(madplay_cmd, "madplay %s &", mp3_path[i]); //保存新要播放的音乐指令
                system(madplay_cmd); 
            }
            else if() //暂停 / 继续
            {

            }
            else if() //下一首
            {
                system("killall -9 madplay");//关闭原来播放的音乐
                i++;
                if(i == 3)
                    i = 0;
                sprintf(madplay_cmd, "madplay %s &", mp3_path[i]); //保存新要播放的音乐指令
                system(madplay_cmd);
            }
            else if() //退出
            {
                system("killall -9 madplay");//关闭原来播放的音乐
                break;
            }
        }
    }
}
```

void ctl_bmp(void)
{
	char *bmp_path[4] = {"1.bmp", "2.bmp", "3.bmp", "4.bmp"};
	int i = 0;
	
	lcd_display_bmp(bmp_path[i], 显示的x坐标，显示y坐标);
	
	while(1)
	{
		//获取滑动方向 / 点击坐标 （滑动控制 / 点击控制）
		if() //上一张
		{
			i--;
            if(i < 0)
            	i = 3;
            lcd_display_bmp(bmp_path[i], 显示的x坐标，显示y坐标);
		}
        else if() //下一张
        {
            i++;
            if(i == 4)
            	i = 0;
            lcd_display_bmp(bmp_path[i], 显示的x坐标，显示y坐标);
        }
        else if() //退出
        {
            break;
        }
	}
}
